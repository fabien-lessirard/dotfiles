+// Helper to parse dates from text
+function parseDate(text) {
+  // Common date formats (DD/MM/YYYY, MM/DD/YYYY, YYYY-MM-DD, DD-MM-YYYY, etc.)
+  const dateRegexes = [
+    /\b(\d{1,2})[./-](\d{1,2})[./-](\d{4})\b/, // DD/MM/YYYY or DD-MM-YYYY
+    /\b(\d{4})[./-](\d{1,2})[./-](\d{1,2})\b/, // YYYY-MM-DD
+    /\b(\d{1,2})[./-](Jan|Feb|Mar|Apr|May|Jun|Jul|Aug|Sep|Oct|Nov|Dec)[./-](\d{4})\b/i, // DD-MON-YYYY
+    /\b(Jan|Feb|Mar|Apr|May|Jun|Jul|Aug|Sep|Oct|Nov|Dec)[./-](\d{1,2})[./-](\d{4})\b/i, // MON-DD-YYYY
+  ];
+
+  for (const regex of dateRegexes) {
+    const match = text.match(regex);
+    if (match) {
+      // Basic parsing, can be improved for ambiguity (e.g., MM/DD vs DD/MM)
+      try {
+        let dateCandidate;
+        if (match[3].length === 4) { // YYYY is present
+          dateCandidate = new Date(`${match[1]}-${match[2]}-${match[3]}`); // Assuming MM-DD-YYYY or DD-MM-YYYY
+        } else { // Assuming DD/MM or MM/DD without year, default to current year
+          const currentYear = new Date().getFullYear();
+          dateCandidate = new Date(`${match[1]}-${match[2]}-${currentYear}`);
+        }
+        
+        // Check if date is valid
+        if (!isNaN(dateCandidate.getTime())) {
+          return dateCandidate.toISOString().split('T')[0]; // Return in YYYY-MM-DD format
+        }
+      } catch (e) {
+        console.warn("Failed to parse date candidate:", match.join('-'), e);
+      }
+    }
+  }
+  return null;
+}

+import React, { useRef, useState, useEffect } from 'react';
+
+const ImageCapture = ({ onCapture }) => {
+    const videoRef = useRef(null);
+    const photoRef = useRef(null);
+    const [hasStream, setHasStream] = useState(false);
+    const [photo, setPhoto] = useState(null);
+
+    const getVideo = () => {
+        navigator.mediaDevices.getUserMedia({
+            video: { width: 400, height: 300 }
+        })
+        .then(stream => {
+            let video = videoRef.current;
+            if (video) {
+                video.srcObject = stream;
+                video.play();
+                setHasStream(true);
+            }
+        })
+        .catch(err => {
+            console.error("Error accessing camera:", err);
+            setHasStream(false);
+        });
+    };
+
+    const takePhoto = () => {
+        let width = 400;
+        let height = 300;
+
+        let video = videoRef.current;
+        let photo = photoRef.current;
+
+        if (photo && video) {
+            let ctx = photo.getContext('2d');
+            photo.width = width;
+            photo.height = height;
+
+            ctx.drawImage(video, 0, 0, width, height);
+            const imageData = photo.toDataURL('image/png');
+            setPhoto(imageData);
+            onCapture(imageData); // Pass the captured image data to the parent component
+        }
+    };
+
+    const clearPhoto = () => {
+        let photo = photoRef.current;
+        if (photo) {
+            let ctx = photo.getContext('2d');
+            ctx.clearRect(0, 0, photo.width, photo.height);
+            setPhoto(null);
+        }
+    };
+
+    useEffect(() => {
+        getVideo();
+        return () => {
+            // Cleanup: stop video stream when component unmounts
+            if (videoRef.current && videoRef.current.srcObject) {
+                videoRef.current.srcObject.getTracks().forEach(track => track.stop());
+            }
+        };
+    }, []);
+
+    return (
+        <div className="image-capture-container">
+            <h3>Capture Product Photo</h3>
+            {hasStream ? (
+                <>
+                    <video ref={videoRef} className="camera-feed"></video>
+                    <button onClick={takePhoto}>Take Photo</button>
+                </>
+            ) : (
+                <p>Camera access denied or not available. Please allow camera access.</p>
+            )}
+
+            <canvas ref={photoRef} style={{ display: 'none' }}></canvas>
+            {photo && (
+                <div className="photo-preview">
+                    <h4>Captured Photo:</h4>
+                    <img src={photo} alt="Captured" />
+                    <button onClick={clearPhoto}>Retake Photo</button>
+                </div>
+            )}
+        </div>
+    );
+};
+
+export default ImageCapture;
