const { initializeApp, cert } = require('firebase-admin/app');
const { getFirestore } = require('firebase-admin/firestore');

// Use a service account
const serviceAccount = require('./sa-key-admin.json');

initializeApp({
  credential: cert(serviceAccount)
});

const db = getFirestore();

const dishes = [
  {
    name: 'Salade de Chèvre Chaud',
    ingredients: ['salade', 'fromage_chevre', 'pain_de_mie']
  },
  {
    name: 'Yaourt aux Fruits',
    ingredients: ['yaourt_grec', 'pommes_bio']
  }
];

async function populateDishes() {
  const batch = db.batch();
  const dishesCollection = db.collection('dishes');

  // Get product IDs for the ingredients
  const productsSnapshot = await db.collection('products').get();
  const productIds = {};
  productsSnapshot.forEach(doc => {
    // A simple way to map product names to IDs. This assumes product names are unique.
    // In a real app, you might have a more robust way of mapping.
    if (doc.data().name.toLowerCase().includes('salade')) {
      productIds['salade'] = doc.id;
    }
    if (doc.data().name.toLowerCase().includes('chèvre')) {
      productIds['fromage_chevre'] = doc.id;
    }
    if (doc.data().name.toLowerCase().includes('pain de mie')) {
      productIds['pain_de_mie'] = doc.id;
    }
    if (doc.data().name.toLowerCase().includes('yaourt')) {
      productIds['yaourt_grec'] = doc.id;
    }
    if (doc.data().name.toLowerCase().includes('pommes')) {
        productIds['pommes_bio'] = doc.id;
    }
  });


  dishes.forEach(dish => {
    const docRef = dishesCollection.doc();
    const ingredientsWithIds = dish.ingredients.map(ingredient => productIds[ingredient]).filter(id => id);
    batch.set(docRef, { ...dish, ingredients: ingredientsWithIds });
  });

  await batch.commit();
  console.log('Dishes collection populated successfully.');
}

populateDishes().catch(console.error);
