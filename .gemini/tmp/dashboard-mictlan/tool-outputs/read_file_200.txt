// --- Helper function to check if user is admin ---
const isAuthenticatedAdmin = async (context) => {
  if (!context.auth) {
    throw new functions.https.HttpsError('unauthenticated', 'The function must be called while authenticated.');
  }

  const uid = context.auth.uid;
  const userDoc = await db.collection('users').doc(uid).get(); // Assuming 'users' collection stores user roles
  if (!userDoc.exists) {
    throw new functions.https.HttpsError('not-found', 'User data not found.');
  }
  const userData = userDoc.data();
  if (userData.role !== 'admin') {
    throw new functions.https.HttpsError('permission-denied', 'Only admin users can perform this action.');
  }
  return true;
};

// --- Clear Products Cloud Function ---
exports.clearProducts = functions.https.onRequest((req, res) => {
  cors(req, res, async () => {
    try {
      // await isAuthenticatedAdmin(context); // Temporarily commented out

      const productsRef = db.collection('products');
      const snapshot = await productsRef.get();
      
      if (snapshot.empty) {
        res.status(200).json({ message: 'No documents found in "products" collection. Nothing to delete.' });
        return;
      }

      const batch = db.batch();
      snapshot.docs.forEach(doc => {
        batch.delete(doc.ref);
      });
      await batch.commit();
      res.status(200).json({ message: `Successfully deleted ${snapshot.docs.length} documents from "products" collection.` });
    } catch (error) {
      console.error("Error in clearProducts:", error);
      res.status(500).send('An error occurred in clearProducts.');
    }
  });
});

// Helper to parse dates from text
function parseDate(text) {
  // Common date formats (DD/MM/YYYY, MM/DD/YYYY, YYYY-MM-DD, DD-MM-YYYY, etc.)
  const dateRegexes = [
    /\b(\d{1,2})[./-](\d{1,2})[./-](\d{4})\b/, // DD/MM/YYYY or DD-MM-YYYY
    /\b(\d{4})[./-](\d{1,2})[./-](\d{1,2})\b/, // YYYY-MM-DD
    /\b(\d{1,2})[./-](Jan|Feb|Mar|Apr|May|Jun|Jul|Aug|Sep|Oct|Nov|Dec)[./-](\d{4})\b/i, // DD-MON-YYYY
    /\b(Jan|Feb|Mar|Apr|May|Jun|Jul|Aug|Sep|Oct|Nov|Dec)[./-](\d{1,2})[./-](\d{4})\b/i, // MON-DD-YYYY
  ];

  for (const regex of dateRegexes) {
    const match = text.match(regex);
    if (match) {
      // Basic parsing, can be improved for ambiguity (e.g., MM/DD vs DD/MM)
      try {
        let dateCandidate;
        if (match[3].length === 4) { // YYYY is present
          dateCandidate = new Date(`${match[1]}-${match[2]}-${match[3]}`); // Assuming MM-DD-YYYY or DD-MM-YYYY
        } else { // Assuming DD/MM or MM/DD without year, default to current year
          const currentYear = new Date().getFullYear();
          dateCandidate = new Date(`${match[1]}-${match[2]}-${currentYear}`);
        }
        
        // Check if date is valid
        if (!isNaN(dateCandidate.getTime())) {
          return dateCandidate.toISOString().split('T')[0]; // Return in YYYY-MM-DD format
        }
      } catch (e) {
        console.warn("Failed to parse date candidate:", match.join('-'), e);
      }
    }
  }
  return null;
}

// --- OCR Cloud Function to extract Expiry Date from Image ---
exports.extractExpiryDateFromImage = functions.https.onCall(async (data, context) => {
  try {
    functions.logger.log('Received data:', data); // Add this line
    // Check authentication if needed (admin or specific user)
    // await isAuthenticatedAdmin(context); // Uncomment if only admins can use OCR

    const base64Image = data.image; // Expecting base64 image data
    if (!base64Image) {
      throw new functions.https.HttpsError('invalid-argument', 'No image data provided.');
    }

    const imageBuffer = Buffer.from(base64Image, 'base64');

    const [result] = await visionClient.textDetection({ // Changed client to visionClient
      image: { content: imageBuffer },
    });
    const detections = result.textAnnotations;

    if (!detections || detections.length === 0) {
      return { expiryDate: null, message: 'No text detected in the image.' };
    }

    const fullText = detections[0].description; // The first annotation is the full text
    functions.logger.log('Full text detected:', fullText);

    const detectedDate = parseDate(fullText);

    if (detectedDate) {
      return { expiryDate: detectedDate, message: 'Expiry date detected.' };
    } else {
      return { expiryDate: null, message: 'No valid expiry date found in the text.' };
    }

  } catch (error) {
    functions.logger.error("Error in extractExpiryDateFromImage:", error);
    throw new functions.https.HttpsError('internal', 'An error occurred during OCR processing.', error.message);
  }
});
