import { useState } from 'react';
import { Settings, RefreshCw, Trash2, PackagePlus } from 'lucide-react';
import { resetDatabase, emptyDatabase } from '../firebase/dbActions';
import Card from './Card';

export default function AdminPanel() {
  const [tableCount, setTableCount] = useState(4);
  const [isUpdating, setIsUpdating] = useState(false);
  const [isEmptying, setIsEmptying] = useState(false);
  const [isPopulatingProducts, setIsPopulatingProducts] = useState(false);
  const [isClearingProducts, setIsClearingProducts] = useState(false);
  const [productFeedback, setProductFeedback] = useState('');
  const [isPopulatingDishes, setIsPopulatingDishes] = useState(false);
  const [dishesFeedback, setDishesFeedback] = useState('');

  const handleReset = async () => {
    setIsUpdating(true);
    try {
      await resetDatabase(tableCount);
    } catch {
      alert("Erreur lors de la mise à jour");
    } finally {
      setIsUpdating(false);
    }
  };

  const handleEmpty = async () => {
    if (!confirm("Êtes-vous sûr de vouloir vider complètement la base de données ? Cette action est irréversible.")) {
      return;
    }
    setIsEmptying(true);
    try {
      await emptyDatabase();
      alert("Base de données vidée avec succès.");
    } catch {
      alert("Erreur lors du vidage de la base de données");
    }
    finally {
      setIsEmptying(false);
    }
  };

  const handlePopulateProducts = async () => {
      setIsPopulatingProducts(true);
      setProductFeedback('Remplissage des produits de test...');
      
      try {
        const response = await fetch('https://us-central1-dashboard-mictlan-staging.cloudfunctions.net/populateProducts', { method: 'POST' });
        if (!response.ok) {
          throw new Error(`HTTP error! status: ${response.status}`);
        }
        const result = await response.json();
        setProductFeedback(result.message || 'Produits de test ajoutés avec succès !');
      } catch (error) {
          console.error("Error populating products:", error);
          setProductFeedback(`Erreur lors du remplissage des produits: ${error.message}`);
      } finally {
          setIsPopulatingProducts(false);
          setTimeout(() => setProductFeedback(''), 3000);
      }
  };

  const handleClearProducts = async () => {
      if (!confirm("Êtes-vous sûr de vouloir vider la collection de produits ? Cette action est irréversible.")) {
          return;
      }
      setIsClearingProducts(true);
      setProductFeedback('Vidage de la collection de produits...');
      
      try {
        const response = await fetch('https://us-central1-dashboard-mictlan-staging.cloudfunctions.net/clearProducts', { method: 'POST' });
        if (!response.ok) {
          throw new Error(`HTTP error! status: ${response.status}`);
        }
        const result = await response.json();
        setProductFeedback(result.message || 'Collection de produits vidée avec succès.');
      } catch (error) {
          console.error("Error clearing products:", error);
          setProductFeedback(`Erreur lors du vidage des produits: ${error.message}`);
      } finally {
          setIsClearingProducts(false);
          setTimeout(() => setProductFeedback(''), 3000);
      }
  };

  const handlePopulateDishes = async () => {
      setIsPopulatingDishes(true);
      setDishesFeedback('Remplissage des plats de test...');
      
      try {
        const response = await fetch('https://us-central1-dashboard-mictlan-staging.cloudfunctions.net/populateDishes', { method: 'POST' });
        if (!response.ok) {
          throw new Error(`HTTP error! status: ${response.status}`);
        }
        const result = await response.json();
        setDishesFeedback(result.message || 'Plats de test ajoutés avec succès !');
      } catch (error) {
          console.error("Error populating dishes:", error);
          setDishesFeedback(`Erreur lors du remplissage des plats: ${error.message}`);
      } finally {
          setIsPopulatingDishes(false);
          setTimeout(() => setDishesFeedback(''), 3000);
      }
  };

  return (
    <Card title="Configuration Système" titleIcon={<Settings />}>
      <div className="flex items-end gap-4 mb-4">
        <div className="flex-1">
          <label className="block text-gray-500 text-[9px] uppercase mb-1">Nombre de tables en salle</label>
          <input 
            type="number" 
            value={tableCount}
            onChange={(e) => setTableCount(e.target.value)}
            className="w-full bg-black border border-gray-800 rounded p-2 text-white outline-none focus:border-mictlan-gold"
            min="1"
            max="50"
          />
        </div>
        
        <button 
          onClick={handleReset}
          disabled={isUpdating}
          className="bg-black/40 text-gray-500 rounded-lg border border-gray-800 hover:text-mictlan-gold transition-all px-4 py-2 font-bold text-[10px] uppercase flex items-center gap-2"
        >
          <RefreshCw className={`w-3 h-3 ${isUpdating ? 'animate-spin' : ''}`} />
          {isUpdating ? 'Mise à jour...' : 'Réinitialiser Salle'}
        </button>
        <button 
          onClick={handleEmpty}
          disabled={isEmptying}
          className="bg-black/40 text-gray-500 rounded-lg border border-gray-800 hover:text-mictlan-gold transition-all px-4 py-2 font-bold text-[10px] uppercase flex items-center gap-2"
        >
          <Trash2 className={`w-3 h-3 ${isEmptying ? 'animate-spin' : ''}`} />
          {isEmptying ? 'Vidage...' : 'Vider DB'}
        </button>
      </div>
      <p className="text-[8px] text-gray-600 mt-2 italic font-mono uppercase">
        * Réinitialiser Salle : Efface toutes les données actuelles pour générer {tableCount} tables. Vider DB : Supprime toutes les données des collections connues (tables, inventory, logs, users). Pour vider toutes les collections inconnues, exécutez `npm run empty-db` or `node empty-db.cjs` dans un terminal.
      </p>

      <div className="flex flex-col gap-4 mb-4 mt-4 border-t border-gray-700 pt-4">
        <label className="block text-sm font-black text-mictlan-gold uppercase tracking-[0.2em] mb-2">
          Gestion des Produits de Test
        </label>
        <div className="flex gap-4">
          <button 
            onClick={handlePopulateProducts}
            disabled={isPopulatingProducts}
            className="flex-1 bg-black/40 text-gray-500 rounded-lg border border-gray-800 hover:text-mictlan-gold transition-all px-4 py-2 font-bold text-[10px] uppercase flex items-center justify-center gap-2"
          >
            <PackagePlus className={`w-3 h-3 ${isPopulatingProducts ? 'animate-spin' : ''}`} />
            {isPopulatingProducts ? 'Remplissage...' : 'Remplir Produits'}
          </button>
          <button 
            onClick={handleClearProducts}
            disabled={isClearingProducts}
            className="flex-1 bg-black/40 text-gray-500 rounded-lg border border-gray-800 hover:text-mictlan-red transition-all px-4 py-2 font-bold text-[10px] uppercase flex items-center justify-center gap-2"
          >
            <Trash2 className={`w-3 h-3 ${isClearingProducts ? 'animate-spin' : ''}`} />
            {isClearingProducts ? 'Vidage...' : 'Vider Produits'}
          </button>
        </div>
        {productFeedback && <p className="text-[10px] text-center text-mictlan-gold mt-2">{productFeedback}</p>}
      </div>

      <div className="flex flex-col gap-4 mb-4 mt-4 border-t border-gray-700 pt-4">
        <label className="block text-sm font-black text-mictlan-gold uppercase tracking-[0.2em] mb-2">
          Gestion des Plats de Test
        </label>
        <div className="flex gap-4">
          <button 
            onClick={handlePopulateDishes}
            disabled={isPopulatingDishes}
            className="flex-1 bg-black/40 text-gray-500 rounded-lg border border-gray-800 hover:text-mictlan-gold transition-all px-4 py-2 font-bold text-[10px] uppercase flex items-center justify-center gap-2"
          >
            <PackagePlus className={`w-3 h-3 ${isPopulatingDishes ? 'animate-spin' : ''}`} />
            {isPopulatingDishes ? 'Remplissage...' : 'Remplir Plats'}
          </button>
        </div>
        {dishesFeedback && <p className="text-[10px] text-center text-mictlan-gold mt-2">{dishesFeedback}</p>}
      </div>
    </Card>
  );
}
